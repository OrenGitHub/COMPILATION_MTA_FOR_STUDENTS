void BubbleSort(IntArray array,int size)
{
    int i := 0;
    int min := -32768;
    int minIndex := -1;
    
    while (i < size)
    {
        int j := i+;
        while (j<size)
        {
            if (array[j] < min)
            {
                int i := array[minIndex];
                array[minIndex] := array[j];
                array[j] := i;
            }
        }
    }
}
